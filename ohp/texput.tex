\[\color{blue}
  \Ham=2\pi\Tr\left(\1half([\aniC,\creC]+1)^2
		 +[\aniC,\phi][\creC,\phi]

\]
\[\color{blue}
 \creC\equiv \crea+i\theta^{\1half}A,\quad
  \aniC\equiv \anih-i\theta^{\1half}\Ab\quad\mbox{(共変微分)}
\]

\[\color{blue}
 C\rightarrow SC\Sd,\quad\phi\rightarrow S\phi\Sd
\]

S=1$}を満たす。\\

\]
